# multiply-function
def multiplyfunc(a,b):
    multiplication = a*b
    print(multiplication)

# Factorial-function
import math

def factorialfunc(a):
    factorial = math.factorial(a)
    print(factorial)

factorialfunc(2)    

# Divide-function
def dividefunc(a,b):
    division = a/b
    print(division)

dividefunc(4,2)
